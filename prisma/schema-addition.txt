model WeeklyMileageSession {
  id              String   @id @default(cuid())
  instructorId    String
  weekStart       DateTime
  weekEnd         DateTime
  startOdometer   Float
  endOdometer     Float
  predictedMiles  Float
  actualMiles     Float
  isComplete      Boolean  @default(false)
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  instructor      InstructorProfile @relation(fields: [instructorId], references: [id], onDelete: Cascade)

  @@index([instructorId])
  @@index([weekStart])
  @@unique([instructorId, weekStart])
}
